# marco MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** App web pessoal de captura → classificação → tarefa/marco → feed, conforme `docs/superpowers/specs/2026-06-11-marco-mvp-design.md`.

**Architecture:** Next.js App Router (Server Components para leitura, Server Actions para mutações) + Supabase (Postgres/Auth/RLS) + 1 route handler para classificação IA (claude-haiku, degrada para manual sem chave). Visual 100% via tokens CSS do DESIGN.md.

**Tech Stack:** Next.js 15, React 19, TypeScript strict, Tailwind v4, shadcn/ui, next-themes, framer-motion, lucide-react, @supabase/ssr, zod, @anthropic-ai/sdk.

### Task 2: Schema Supabase via MCP

- [ ] `mcp__supabase__apply_migration` nome `marco_mvp_schema`:

```sql
create table public.projects (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  name text not null,
  slug text not null,
  color_key text not null default 'neutral'
    check (color_key in ('ae','kairos','esca7','busca-clt','neutral')),
  archived boolean not null default false,
  created_at timestamptz not null default now(),
  unique (user_id, slug)
);

create table public.tasks (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  project_id uuid references public.projects(id) on delete set null,
  title text not null,
  note text,
  status text not null default 'backlog' check (status in ('backlog','doing','done')),
  starred boolean not null default false,
  created_at timestamptz not null default now(),
  done_at timestamptz
);

create table public.milestones (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  project_id uuid references public.projects(id) on delete set null,
  task_id uuid references public.tasks(id) on delete set null,
  title text not null,
  note text,
  source text not null default 'manual' check (source in ('manual','task','inbox')),
  occurred_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

create table public.inbox_items (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references auth.users(id) on delete cascade,
  raw_text text not null,
  status text not null default 'pending' check (status in ('pending','converted','discarded')),
  suggestion jsonb,
  created_at timestamptz not null default now()
);

create index tasks_user_status_idx on public.tasks (user_id, status);
create index milestones_user_occurred_idx on public.milestones (user_id, occurred_at desc);
create index inbox_user_status_idx on public.inbox_items (user_id, status);
create index tasks_project_idx on public.tasks (project_id);
create index milestones_project_idx on public.milestones (project_id);
create index milestones_task_idx on public.milestones (task_id);

alter table public.projects enable row level security;
alter table public.tasks enable row level security;
alter table public.milestones enable row level security;
alter table public.inbox_items enable row level security;

create policy "own projects" on public.projects for all
  using ((select auth.uid()) = user_id) with check ((select auth.uid()) = user_id);
create policy "own tasks" on public.tasks for all
  using ((select auth.uid()) = user_id) with check ((select auth.uid()) = user_id);
create policy "own milestones" on public.milestones for all
  using ((select auth.uid()) = user_id) with check ((select auth.uid()) = user_id);
create policy "own inbox" on public.inbox_items for all
  using ((select auth.uid()) = user_id) with check ((select auth.uid()) = user_id);

create or replace function public.ensure_seed_projects()
returns void language plpgsql security invoker as $$
begin
  if not exists (select 1 from public.projects where user_id = auth.uid()) then
    insert into public.projects (user_id, name, slug, color_key) values
      (auth.uid(), 'AE', 'ae', 'ae'),
      (auth.uid(), 'Kairós', 'kairos', 'kairos'),
      (auth.uid(), 'Esca7', 'esca7', 'esca7'),
      (auth.uid(), 'Busca CLT', 'busca-clt', 'busca-clt');
  end if;
end $$;
```
